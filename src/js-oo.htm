    constructor(){
        this.data = new Date();
        this.quantidade = 1;
        this.valor = 5.5;
    }
}

contructor, method 
class Negociacao{
    constructor(d, q, v){
        this.data = d;
        this.quantidade = q;
        this.valor = v;
    }

    obterVolume(){
        return this.quantidade * this.valor;
    }
}

var n1 = new Negociacao(new Date(), 1, 6.6);
console.log(n1.valor) //--6.6
convenção de encapsulamento, private freeze(this) 
class Negociacao{
    constructor(d, q, v){
        this._data = d;
        this._quantidade = q;
        this._valor = v;

        Object.freeze(this);
    }

    obterVolume(){
        return this._quantidade * this._valor;
    }
}

var n1 = new Negociacao(new Date(), 1, 6.6);
n1._valor = 1.1; //--não funciona
console.log(n1._valor)
